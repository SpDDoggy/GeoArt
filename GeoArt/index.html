使用GeoPandas绘制世界人口密度图

import geopandas as gpd
import matplotlib.pyplot as plt
from mpl_toolkits.axes_grid1 import make_axes_locatable

# Read geographic data
world = gpd.read_file(gpd.datasets.get_path('naturalearth_lowres'))

# Calculate population density (adjusting area unit for better scale)
world['density'] = world['pop_est'] / world['area_sqkm'] # Assuming 'area_sqkm' exists or calculate if needed

# Create figure and axes
fig, ax = plt.subplots(1, 1, figsize=(14, 9))

# Set color map and normalization
cmap = 'YlOrRd' # Yellow-Orange-Red for density
vmin, vmax = world['density'].min(), world['density'].max()

# Plot the map
divider = make_axes_locatable(ax)
cax = divider.append_axes("right", size="3%", pad=0.1) # Thinner color bar
world.plot(column='density', cmap=cmap, linewidth=0.5, ax=ax, edgecolor='0.7', legend=True, cax=cax, 
           legend_kwds={'label': "Population Density (people/km²)", 'orientation': "vertical"})

# Set title and labels
ax.set_title('World Population Density Distribution Map', fontsize=18, color='#333')
ax.set_xlabel('Longitude')
ax.set_ylabel('Latitude')

# Add grid lines
ax.grid(True, linestyle=':', alpha=0.6, color='gray')

# Remove axis ticks
ax.set_xticks([])
ax.set_yticks([])

# Adjust layout
plt.tight_layout()

# Display plot
plt.show()

# Save plot (optional)
# plt.savefig('world_population_density.png', dpi=300, bbox_inches='tight')

代码说明

这段Python代码利用GeoPandas库读取全球地理数据，并计算各国的人口密度。通过Matplotlib，它生成了一个色彩鲜明的世界人口密度专题地图。地图通过颜色深浅直观地展示了不同区域的人口分布差异，并包含了图例、标题和网格线，提高了地图的专业性和可读性。

使用PySAL进行空间自相关分析（Moran's I）

import geopandas as gpd
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from esda.moran import Moran
from libpysal.weights import Queen, KNN
from splot.esda import moran_scatterplot, lisa_cluster

# Load data (replace 'neighborhoods.shp' with your actual data path)
try:
    gdf = gpd.read_file('neighborhoods.shp')
except Exception as e:
    print(f"Error loading shapefile: {e}. Using dummy data for demonstration.")
    # Create dummy GeoDataFrame if file not found
    from shapely.geometry import Point
    data = {'id': range(5), 'median_income': np.random.rand(5) * 100000 + 30000}
    geometry = [Point(i, i) for i in range(5)]
    gdf = gpd.GeoDataFrame(data, geometry=geometry, crs="EPSG:4326")
    # For dummy data, make sure it's valid for Queen weights or use KNN
    if len(gdf) > 1:
        # Create a simple convex hull to ensure connectivity if needed for Queen weights
        gdf_buffered = gdf.buffer(0.01)
        gdf = gpd.GeoDataFrame(gdf, geometry=gdf_buffered.centroid, crs="EPSG:4326")
    print("Dummy GeoDataFrame created.")


# Ensure consistent CRS (e.g., for area calculation or specific analysis)
if gdf.crs is None or not gdf.crs.equals("EPSG:3857"):
    gdf = gdf.to_crs(epsg=3857)
    print("CRS transformed to EPSG:3857.")

# Calculate area if needed (example conversion to square kilometers)
# For real data, 'area' attribute might be directly available or calculated based on appropriate CRS
if 'area_sqkm' not in gdf.columns:
    gdf['area_sqkm'] = gdf.area / 1_000_000 # Convert m^2 to km^2
    print("Area calculated.")

# Create spatial weights matrix (Queen adjacency for polygons, or KNN for points)
if gdf.geom_type.iloc[0] in ['Polygon', 'MultiPolygon']:
    w = Queen.from_dataframe(gdf)
    print("Queen weights created.")
else:
    # Fallback to KNN for point data or disconnected polygons for demonstration
    w = KNN.from_dataframe(gdf, k=min(len(gdf)-1, 5)) # Use 5 nearest neighbors
    print(f"KNN weights (k={min(len(gdf)-1, 5)}) created.")

# Row-standardize weights
w.transform = 'r'
print("Weights row-standardized.")

# Select variable for analysis
variable = 'median_income'
if variable not in gdf.columns:
    print(f"Variable '{variable}' not found. Please ensure your data has this column.")
    # Fallback to a dummy variable if not found
    gdf[variable] = np.random.rand(len(gdf)) * 1000 # Dummy data
    print(f"Dummy variable '{variable}' created.")

# Calculate Moran's I index
try:
    moran = Moran(gdf[variable], w)
    print(f"Moran's I value: {moran.I:.4f}")
    print(f"Z-score: {moran.z_norm:.4f}")
    print(f"p-value: {moran.p_norm:.4f}")
except Exception as e:
    print(f"Error calculating Moran's I: {e}. Check if data is suitable for spatial weights.")
    moran = None # Ensure moran is None if calculation fails

# Create Moran scatter plot
if moran:
    fig, ax = moran_scatterplot(moran, aspect_equal=True)
    ax.set_title('Moran Scatter Plot', fontsize=16)
    ax.set_xlabel(f'{variable} (Standardized)')
    ax.set_ylabel('Spatial Lagged Value')
    plt.tight_layout()
    plt.show()

# LISA cluster map
if moran and moran.p_sim is not None: # Check if permutation p-values exist for LISA
    fig, ax = plt.subplots(figsize=(10, 8))
    lisa_cluster(moran, gdf, p=0.05, ax=ax, figsize=(10, 8)) # Use p-value for significance
    ax.set_title('LISA Cluster Map (p < 0.05)', fontsize=16)
    ax.set_axis_off() # Remove axis for cleaner map
    plt.tight_layout()
    plt.show()
elif moran:
    print("LISA cluster map could not be generated. Permutation p-values for Moran's I might be missing or insufficient.")

代码说明

此代码演示了如何使用PySAL库执行空间自相关分析。它首先加载地理数据并构建空间权重矩阵，然后计算Moran's I指数以量化空间依赖性。代码进一步生成了Moran散点图和局部空间自相关（LISA）聚类图，这些图表有助于识别数据中的空间热点、冷点以及异常值，为空间模式分析提供可视化支持。
